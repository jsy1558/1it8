import folium               #지도를 그려주는 패키지
import pandas as pd
import geopandas as gpd     #pandas 패키지를 기반으로 한 공간 데이터 처리용 패키지
import numpy as np

#시군구 지도(sig.json) 파일 불러오기
geo_sig=gpd.read_file('sig.json')                               
#전국 시군구 경계 지도 데이터 불러오기
geo_sig=geo_sig[geo_sig['SIG_CD'].astype(int).between(11110,11740)]            
#그 중에서 서울 데이터만 남기기

#읍면동 지도(emd.json) 파일 불러오기
geo_emd=gpd.read_file('emd.json')                               
#전국 읍면동 경계 지도 데이터 불러오기
geo_emd=geo_emd[geo_emd['EMD_CD'].astype(int).between(11110101,11740110)]            
#그 중에서 서울 데이터만 남기기	
geo_emd['AREA_CODE']=geo_emd['EMD_CD'].map(str)+"00"            
#법정동 코드 컬럼명 10자리로 통일하기
geo_emd['SIG_CD']=geo_emd['EMD_CD'].str[:5].map(str)           
#EMD_CD에서 시군구 코드 추출하기
geo_emd=geo_emd[['SIG_CD','EMD_CD','AREA_CODE','EMD_ENG_NM','EMD_KOR_NM','geometry']] 
#컬럼 순서 정렬
geo_emd_json=geo_emd.to_json()

#folium.Choropleth 함수를 사용하려면 숫자형 데이터가 필요해서 변환함(명목형 데이터 사용 불가)

map_target1=folium.Map(location=[37.56,127],      #서울 좌표

folium.Choropleth(geo_data=geo_emd_json,        #지도 데이터
									data=target_areas1_data,      #지도에 표시할 데이터
									columns=['AREA_CODE', 'COLOR_CODE'],
									key_on='feature.properties.AREA_CODE',
									fill_color='PuBu',            #해당하는 지역은 PuBu 계열로 칠함
									nan_fill_color='white',       #해당하지 않는 지역은 하얀색으로 덮음
									fill_opacity=0.5,             #투명도 조절(보다 연하게)
									line_opacity=0.2,             #지도의 선 굵기 조절
									).add_to(map_target1)
